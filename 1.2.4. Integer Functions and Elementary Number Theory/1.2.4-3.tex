$\lfloor x \rfloor = \max \{ m \in \mathbb{Z} | m \leq x \}$

$\lceil x \rceil = \min \{ m \in \mathbb{Z} | m \geq x \}$

\subsubsection*{(a)}

$\lfloor x \rfloor < n \implies n > \max \{ m \in \mathbb{Z} | m \leq x \}  \implies x < n$

$x < n \implies \lfloor x \rfloor \leq x < n \implies \lfloor x \rfloor < n$

\subsubsection*{(b)}

$n \leq \lfloor x \rfloor \implies n \leq \lfloor x \rfloor \leq x \implies n \leq x$

$n \leq x \implies n \leq \max \{ m \in \mathbb{Z} | m \leq x \} = \lfloor x \rfloor \implies n \leq \lfloor x \rfloor$

\subsubsection*{(c)}

$\lceil x \rceil \leq n \implies x \leq \lceil x \rceil \leq n \implies x \leq n$

$x \leq n \implies n \geq \min \{ m \in \mathbb{Z} | m \geq x \} = \lceil x \rceil \implies \lceil x \rceil \leq n$

\subsubsection*{(d)}

$n < \lceil x \rceil \implies n < \min \{ m \in \mathbb{Z} | m \geq x \} \implies n < x$

$n < x \implies n < x \leq \lceil x \rceil \implies n < \lceil x \rceil$

\subsubsection*{(e)}

$\lfloor x \rfloor = n \implies n \leq x, n + 1 \nleq x \implies n \leq x, n + 1 > x \implies n \leq x, n > x - 1 \implies x - 1 < n \leq x$

$x-1 < n \leq x \implies n \leq x, x < n + 1 \implies n \leq x < n + 1$

$n \leq x < n + 1 \implies n \leq x, n+1 \nleq x \implies n = \lfloor x \rfloor$

\subsubsection*{(f)}

$\lceil x \rceil = n \implies n \geq x, n - 1 \ngeq x \implies n \geq x, n - 1 < x \implies n \geq x, n < x + 1 \implies x \leq n < x + 1$

$x \leq n < x + 1 \implies x \leq n, x > n - 1 \implies n - 1 < x \leq n$

$n - 1 < x \leq n \implies n \geq x, n-1 \ngeq x \implies n = \lceil x \rceil$
