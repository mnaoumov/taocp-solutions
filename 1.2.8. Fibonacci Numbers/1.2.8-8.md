# 1.2.8-8

${} F_{-1} = F_1 - F_0 = 1-0 = 1 {}$

${} F_{-2} = F_0 - F_{-1} = 0-1 = -1 {}$

${} F_{-3} = F_{-1} - F_{-2} = 1 - (-1) = 2 {}$

Let's prove ${} F_{-n} = (-1)^{n+1}F_n {}$

${} n = 1: F_{-1} = (-1)^{1+1}F_1 {}$

${} (n \implies n + 1) {}$

${} F_{-(n+1)} = F_{-n-1} = F_{-n+1} - F_{-n} = F_{-(n-1)} - F_{-n} = (-1)^{(n-1)+1}F_{n-1} - (-1)^{n+1}F_n = {}$

${} = (-1)^{(n+1)+1}(F_n + F_{n-1}) = (-1)^{(n+1)+1}(F_n + F_{n-1}) = (-1)^{(n+1)+1} F_{n+1} {}$
