$(\sum_{j=1}^n a_jx_j)(\sum_{j=1}^n b_jy_j) = \sum_{1 \leq j,k \leq n} a_jb_kx_jy_k =: X$

$(\sum_{j=1}^n a_jy_j)(\sum_{j=1}^n b_jx_j) = \sum_{1 \leq j,k \leq n} a_jb_kx_ky_j =: Y$

$A := \sum_{1 \leq j, k \leq n } (a_jb_k - a_kb_j)(x_jy_k - x_ky_j) = \overbrace{\sum_{1 \leq j < k \leq n } (a_jb_k - a_kb_j)(x_jy_k - x_ky_j)}^B + \overbrace{\sum_{1 \leq k < j \leq n } (a_jb_k - a_kb_j)(x_jy_k - x_ky_j)}^C + \overbrace{\sum_{1 \leq j = k \leq n } (a_jb_k - a_kb_j)(x_jy_k - x_ky_j)}^D$

$C = \sum_{1 \leq j < k \leq n } (a_kb_j - a_jb_k)(x_ky_j - x_jy_k) = \sum_{1 \leq j < k \leq n } (a_jb_k - a_kb_j)(x_jy_k - x_ky_j) = B$

$D = \sum_{i=1}^n(a_jb_j-a_jb_j)(x_jy_j-x_jy_j) = 0$

$A = 2B$

$A = \sum_{1\leq j,k \leq n} (a_jb_kx_jy_k - a_jb_kx_ky_j - a_kb_jx_jy_k + a_kb_jx_ky_j) = (\sum_{j=1}^n a_jx_j)(\sum_{k=1}^n b_ky_k) - (\sum_{j=1}^n a_jy_j)(\sum_{k=1}^n b_kx_k) - (\sum_{j=1}^n b_jx_j)(\sum_{k=1}^n a_ky_k) + (\sum_{j=1}^n b_jy_j)(\sum_{k=1}^n a_kx_k) = X - Y - Y + X = 2(X-Y)$

$\implies 2(X-Y) = B \implies X = Y + B$
