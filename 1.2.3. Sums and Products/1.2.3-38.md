# 1.2.3-38

${} C = C(x_1,x_2,\ldots,x_n,y_1,y_2,\ldots,y_n) = \left(\begin{matrix} \frac{1}{x_1+y_1} & \frac{1}{x_1+y_2} & \cdots & \frac{1}{x_1+y_n} \\ \frac{1}{x_2+y_1} & \frac{1}{x_2+y_2} & \cdots & \frac{1}{x_2+y_n} \\ \vdots & \vdots & \ddots & \vdots \\ \frac{1}{x_n+y_1} & \frac{1}{x_n+y_2} & \cdots & \frac{1}{x_n+y_n} \end{matrix}\right) {}$

${} |C| \overset{\substack{R_2 \leftarrow R_2 - R_1\\R_3 \leftarrow R_3 - R_1 \\ \vdots \\ R_n \leftarrow R_n - R_1 \\ \frac{1}{x_k + y_j} - \frac{1}{x_i + y_j} = \frac{x_i-x_k}{(x_k + y_j)(x_i + y_j)}}}{=} \left|\begin{matrix} \frac{1}{x_1+y_1} & \frac{1}{x_1+y_2} & \cdots & \frac{1}{x_1+y_n} \\ \frac{x_1-x_2}{(x_2+y_1)(x_1+y_1)} & \frac{x_1-x_2}{(x_2+y_2)(x_1+y_2)} & \cdots & \frac{x_1-x_2}{(x_2+y_n)(x_1+y_n)} \\ \vdots & \vdots & \ddots & \vdots \\ \frac{x_1-x_n}{(x_n+y_1)(x_1+y_1)} & \frac{x_1-x_n}{(x_n+y_2)(x_1+y_2)} & \cdots & \frac{x_n-x_2}{(x_n+y_n)(x_1+y_n)} \end{matrix}\right| \overset{\substack{C_1 \leftarrow C_1 (x_1 + y_1) \\ C_2 \leftarrow C_2 (x_1 + y_2) \\ \vdots \\ C_n \leftarrow C_n (x_1 + y_n) }}{=} \prod_{j=1}^n\frac{1}{x_1+y_j} \left|\begin{matrix} 1 & 1 & \cdots & 1 \\ \frac{x_1-x_2}{x_2+y_1} & \frac{x_1-x_2}{x_2+y_2} & \cdots & \frac{x_1-x_2}{x_2+y_n} \\ \vdots & \vdots & \ddots & \vdots \\ \frac{x_1-x_n}{x_n+y_1} & \frac{x_1-x_n}{x_n+y_2} & \cdots & \frac{x_n-x_2}{x_n+y_n} \end{matrix}\right| = {}$

${} \overset{\substack{R_1 \leftarrow \frac{R_1}{x_1-x_2} \\ R_2 \leftarrow \frac{R_2}{x_1-x_3} \\ \vdots \\ R_n \leftarrow \frac{R_n}{x_1-x_n} }}{=} \frac{1}{x_1+y_1}\prod_{j=2}^n\frac{x_1 - x_j}{x_1+y_j} \left|\begin{matrix} 1 & 1 & \cdots & 1 \\ \frac{1}{x_2+y_1} & \frac{1}{x_2+y_2} & \cdots & \frac{1}{x_2+y_n} \\ \vdots & \vdots & \ddots & \vdots \\ \frac{1}{x_n+y_1} & \frac{1}{x_n+y_2} & \cdots & \frac{1}{x_n+y_n} \end{matrix}\right| \overset{\substack{C_2 \leftarrow C_2 - C_1 \\ C_3 \leftarrow C_3 - C_1 \\ \vdots \\ C_n \leftarrow C_n - C_1 \\ \frac{1}{x_i + y_j} - \frac{1}{x_i + y_l} = \frac{y_l-y_j}{(x_i + y_j)(x_i + y_l)}}}{=} \frac{1}{x_1+y_1}\prod_{j=2}^n\frac{x_1-x_j}{x_1+y_j} \left|\begin{matrix} 1 & 0 & \cdots & 0 \\ \frac{1}{x_2+y_1} & \frac{y_1-y_2}{(x_2+y_2)(x_2+y_1)} & \cdots & \frac{y_1-y_n}{(x_2+y_n)(x_2+y_1)} \\ \vdots & \vdots & \ddots & \vdots \\ \frac{1}{x_n+y_1} & \frac{y_1-y_2}{(x_n+y_2)(x_n+y_1)} & \cdots & \frac{y_1-y_n}{(x_n+y_n)(x_n+y_1)} \end{matrix}\right| = {}$

${} \overset{\substack{R_2 \leftarrow R_2 \cdot (x_2 + y_1) \\ R_3 \leftarrow R_3 \cdot (x_3 + y_1) \\ \vdots \\ R_n \leftarrow R_n \cdot (x_n + y_1)}}{=} \frac{1}{x_1+y_1}\prod_{j=2}^n\frac{x_1-x_j}{x_1+y_j} \prod_{i=2}^n\frac{1}{x_i + y_1}\left|\begin{matrix} 1 & 0 & \cdots & 0 \\ 1 & \frac{y_1-y_2}{x_2+y_2} & \cdots & \frac{y_1-y_n}{x_2+y_n} \\ \vdots & \vdots & \ddots & \vdots \\ 1 & \frac{y_1-y_2}{x_n+y_2} & \cdots & \frac{y_1-y_n}{x_n+y_n} \end{matrix}\right| \overset{\substack{C_2 \leftarrow \frac{C_2}{y_1-y_2} \\ C_3 \leftarrow \frac{C_3}{y_1-y_3} \\ \vdots \\ C_n \leftarrow \frac{C_n}{y_1-y_n}}}{=} \frac{1}{x_1+y_1}\prod_{j=2}^n\frac{x_1-x_j}{x_1+y_j} \prod_{i=2}^n\frac{y_1-y_i}{x_i + y_1} \left|\begin{matrix} 1 & 0 & \cdots & 0 \\ 1 & \frac{1}{x_2+y_2} & \cdots & \frac{1}{x_2+y_n} \\ \vdots & \vdots & \ddots & \vdots \\ 1 & \frac{1}{x_n+y_2} & \cdots & \frac{1}{x_n+y_n} \end{matrix}\right| = {}$

${} \overset{\text{Expand by}\ R_1}{=} \frac{1}{x_1+y_1}\prod_{j=2}^n\frac{x_j-x_1}{x_1+y_j} \prod_{i=2}^n\frac{y_i-y_1}{x_i + y_1} |C(x_2,y_2,x_3,y_3,\ldots,x_n,y_n)| {}$

${} |C(x_1,y_1)| = \frac{1}{x_1+y_1} {}$

By induction

${} |C| = \prod_{i=1}^n\frac{1}{x_i+y_i} \prod_{1 \leq i \leq j \leq n} \frac{(x_j-x_i)(y_j-y_i)}{(x_i+y_j)(x_j + y_i)} = \prod_{1 \leq i,j \leq n}\frac{1}{x_i+y_j} \prod_{1 \leq i < j \leq n} (x_j-x_i)(y_j-y_i) {}$
