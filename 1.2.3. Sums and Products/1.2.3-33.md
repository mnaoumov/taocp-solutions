# 1.2.3-33

${} f_{nr} (x_1,x_2,\ldots,x_n) := \sum_{j=1}^{n}\frac{x_j^r}{\prod_{1 \leq k \leq n, k \neq j}(x_j-x_k)} {}$

We need to prove

${} f_{nr}(x_1,x_2,\ldots,x_n) = 0, 0 \leq r \leq n - 2 {}$

${} f_{n,n-1}(x_1,x_2,\ldots,x_n) = 1, r = n - 1 {}$

${} f_{nn}(x_1,x_2,\ldots,x_n) = x_1 + x_2 + \cdots + x_n, r = n {}$

Induction by ${} n {}$

${} n = 1 {}$

${} f_{n0}(x_1) = x_1^0 = 1, r = 0 = n - 1 {}$

${} f_{n1}(x_1) = x_1^1 = x_1, r = 1 = n {}$

${} (n \implies n + 1) {}$

${} f_{n+1,r}(x_1,x_2,\ldots,x_n,x_{n+1}) = \sum_{j=1}^{n+1}\frac{x_j^r}{\prod_{1 \leq k \leq n + 1, k \neq j}(x_j-x_k)} = \sum_{j=1}^{n+1}\frac{x_j^r}{\prod_{1 \leq k \leq n - 1, k \neq j}(x_j-x_k) \cdot \prod_{n \leq k \leq n + 1, k \neq j}(x_j-x_k) } = \sum_{j=1}^{n-1}\frac{x_j^r}{\prod_{1 \leq k \leq n - 1, k \neq j}(x_j-x_k) \cdot (x_j-x_n)(x_j-x_{n+1}) } + \frac{x_n^r}{\prod_{1 \leq k \leq n - 1}(x_n-x_k)\cdot(x_n-x_{n+1})} + \frac{x_{n+1}^r}{\prod_{1 \leq k \leq n - 1}(x_{n+1}-x_k)\cdot(x_{n+1}-x_{n})} =: A {}$

${} \frac{1}{(x_j-x_n)(x_j-x_{n+1})} = \frac{1}{x_{n}-x_{n+1}} (\frac{1}{x_j-x_n} - \frac{1}{x_j-x_{n+1}}) {}$

${} A = \frac{1}{x_n-x_{n+1}}(\sum_{j=1}^{n-1}\frac{x_j^r}{\prod_{1 \leq k \leq n - 1, k \neq j}(x_j-x_k) \cdot (x_j-x_n)} - \sum_{j=1}^{n-1}\frac{x_j^r}{\prod_{1 \leq k \leq n - 1, k \neq j}(x_j-x_k) \cdot (x_j-x_{n+1})} + \frac{x_n^r}{\prod_{1 \leq k \leq n - 1}(x_n-x_k)} - \frac{x_{n+1}^r}{\prod_{1 \leq k \leq n - 1}(x_{n+1}-x_k)}) =: B {}$

${} x'_1 := x_1, x'_2 := x_2, \ldots, x'_{n-1} := x_{n-1}, x'_n := x_{n+1} {}$

${} B = \frac{1}{x_n-x_{n+1}}(\sum_{j=1}^{n}\frac{x_j^r}{\prod_{1 \leq k \leq n, k \neq j}(x_j-x_k)} - \sum_{j=1}^{n}\frac{(x'_j)^r}{\prod_{1 \leq k \leq n, k \neq j}(x'_j-x'_k)}) = \frac{f_{nr}(x_1,x_2,\ldots,x_{n-1},x_n) - f_{nr}(x'_1,x'_2,\ldots,x'_{n-1},x'_n)}{x_n-x_{n+1}} {}$

${} 0 \leq r \leq n - 2: f_{n+1,r}(x_1,x_2,\ldots,x_n,x_{n+1}) = \frac{0 - 0}{x_n-x_{n+1}} = 0 {}$

${} r = n - 1 = (n + 1) - 2: f_{n+1,n - 1}(x_1,x_2,\ldots,x_n,x_{n+1}) = \frac{1 - 1}{x_n-x_{n+1}} = 0 {}$

${} r = n = (n + 1) - 1: f_{n+1,n}(x_1,x_2,\ldots,x_n,x_{n+1}) = \frac{(x_1 + x_2 + \cdots x_{n-1} + x_n) - (x'_1 + x'_2 + \cdots x'_{n-1} + x'_n)}{x_n-x_{n+1}} = \frac{x_n-x_{n+1}}{x_n-x_{n+1}} = 1 {}$

${} r = n + 1 {}$

${} p(t) := (t-x_1)(t-x_2) \cdot\ \cdots\ \cdot (t-x_n)(t-x_{n+1}) {}$

${} f_{n+1,r}(x_1,x_2,\ldots,x_n,x_{n+1}) = \sum_{j=1}^{n+1} \left.\frac{t^r}{\frac{p(t)}{t-x_j}}\right|_{t = x_j} {}$

${} p(x_j) = 0 \implies \left.\frac{p(t)}{t-x_j}\right|_{t=x_j} = \left.\frac{p(t)-p(x_j)}{t-x_j}\right|_{t=x_j} = p'(x_j) {}$

${} f_{n+1,r}(x_1,x_2,\ldots,x_n,x_{n+1}) = \sum_{j=1}^{n+1} \left.\frac{t^r}{p'(t)}\right|_{t = x_j} {}$

${} p(t)=t^{n+1} - (\sum_{j=1}^{n+1} x_j) \cdot t^n + u(t), \deg u \leq n-1 {}$

${} S := \sum_{j=1}^{n+1} x_j {}$

${} p'(t) = (n+1)t^n - nSt^{n-1} + u'(t), \deg u' \leq n-2 {}$

${} f_{n+1,n+1}(x_1,x_2,\ldots,x_n,x_{n+1}) = \sum_{j=1}^{n+1} \left.\frac{t^{n+1}}{p'(t)}\right|_{t = x_j} {}$

${} \deg t^{n+1} = n + 1, \deg p' = n {}$

${} t^{n+1} = (at+b)p'(t) + r(t), \deg r \leq n - 1 {}$

${} r(t) = \sum_{i=0}^{n-1}r_it^i {}$

${} f_{n+1,n+1}(x_1,x_2,\ldots,x_n,x_{n+1}) = \sum_{j=1}^{n+1} \left.\frac{t^{n+1}}{p'(t)}\right|_{t = x_j} = \sum_{j=1}^{n+1} \left.(at + b + \sum_{i=0}^{n-1}\frac{r_i t^{i}}{p'(t)})\right|_{t = x_j} = a \sum_{j=1}^{n+1} x_j + b \sum_{j=1}^{n+1} 1 + \sum_{i=0}^{n-1} r_i f_{n+1,i}(x_1,x_2,\ldots,x_n,x_{n+1}) = aS + (n+1)b + 0 = aS + (n+1)b {}$

${} t^{n+1} = (at+b)p'(t) + r(t) = (at+b)((n+1)t^n - nSt^{n-1} + u'(t)) + r(t) = (n+1)at^{n+1} - naSt^n + atu'(t) + b(n+1)t^n - bnSt^{n-1} + bu'(t) + r(t) = {}$

${} = (n+1)at^{n+1} + t^n(-naS+b(n+1)) + \overbrace{(atu'(t)-bnSt^{n-1} + bu'(t) + r(t))}^{v(t)}, \deg v \leq n - 1 {}$

${} \implies 1 = (n+1)a, 0 = -naS + b(n+1) {}$

${} a = \frac{1}{n+1}, b = \frac{nS}{(n+1)^2} {}$

${} \implies f_{n+1,n+1}(x_1,x_2,\ldots,x_n,x_{n+1}) = aS + (n+1)b = \frac{S}{n+1} + \frac{nS}{n+1} = S {}$

### v2

${} g_r(z) := \frac{z^r}{(z-x_1)(z-x_2) \cdot\ \cdots\ \cdot (z-x_n)} {}$

${} z=x_j {}$ - pole of ${} g_r(z) {}$

${} \mathrm{res}_{x_j}g_r(z) = \lim_{z\to x_j} g_r(z)(z-x_j) {}$

${} f_{nr} (x_1,x_2,\ldots,x_n) := \sum_{j=1}^{n}\frac{x_j^r}{\prod_{1 \leq k \leq n, k \neq j}(x_j-x_k)} = \sum_{j=1}^n (\mathrm{res}_{x_j}g_r(z)) = \frac{1}{2\pi i}\oint_{|z|<R}g_r(z)dz {}$, where ${} R > \max\lbrace|x_1|,|x_2|,\ldots,|x_n|\rbrace {}$

${} g_r(z) = \frac{z^{r-n}}{(1-\frac{x_1}{z})(1-\frac{x_2}{z}) \cdot\ \cdots\ \cdot (1-\frac{x_n}{z})} = z^{r-n} (\sum_{k_1=0}^\infty(\frac{x_1}{z})^{k_1}) \cdot\ \cdots\ \cdot (\sum_{k_n=0}^\infty(\frac{x_n}{z})^{k_n}) = z^{r-n} \sum_{k_1, k_2, \ldots, k_n \geq 0} z^{-(k_1 + k_2 + \cdots + k_n)}x_1^{k_1} \cdot\ \cdots\ \cdot x_n^{k_n} {}$

${} \oint_{|z|<R}g_r(z)dz = 2 \pi i \cdot ([z^{-1}]g_r(z)) {}$

${} f_{nr} (x_1,x_2,\ldots,x_n) = [z^{-1}]g_r(z) = \sum_{\substack{k_1,k_2,\ldots,k_n \geq 0 \\ k_1 + k_2 + \cdots + k_n = r - n + 1}} x_1^{k_1} \cdot\ \cdots\ \cdot x_n^{k_n} {}$

${} 0 \leq r \leq n - 2 \implies r - n + 1 < 0 \implies {}$ such indices in sum don't exist ${} \implies f_{nr} (x_1,x_2,\ldots,x_n) = 0 {}$

${} r = n - 1 \implies f_{nr} (x_1,x_2,\ldots,x_n) = \sum_{k_1=k_2=\cdots=k_n = 0} x_1^{k_1} \cdot\ \cdots\ \cdot x_n^{k_n} = 1 {}$

${} r = n - 1 \implies f_{nr} (x_1,x_2,\ldots,x_n) = \sum_{\substack{k_1,k_2,\ldots,k_n \geq 0\\k_1 + k_2 + \cdots + k_n = 1}} x_1^{k_1} \cdot\ \cdots\ \cdot x_n^{k_n} = x_1 + x_2 + \ldots + x_n {}$
