# 1.2.3-41

${} C=C(x_1,x_2,\ldots,x_n,y_1,y_2,\ldots,y_n) = \left(\begin{matrix} \frac{1}{x_1+y_1} & \frac{1}{x_1+y_2} & \cdots & \frac{1}{x_1+y_n} \\ \frac{1}{x_2+y_1} & \frac{1}{x_2+y_2} & \cdots & \frac{1}{x_2+y_n} \\ \vdots & \vdots & \ddots & \vdots \\ \frac{1}{x_n+y_1} & \frac{1}{x_n+y_2} & \cdots & \frac{1}{x_n+y_n} \end{matrix}\right) {}$

${} (C^{-1})_{ij} = \frac{(-1)^{i+j}\hat{C}_{ji}}{|C|} {}$

${} |C| = \prod_{1 \leq k,l \leq n}\frac{1}{x_k+y_l} \prod_{1 \leq k < l \leq n} (x_l-x_k)(y_l-y_k) {}$

${} \hat{C}_{ji} = |C(x'_1,y'_1,x'_2,y'_2,\ldots,x'_{n-1},y'_{n-1})| {}$

${} (x'_1,x'_2,\ldots,x'_{n-1}) = (x_1,x_2,\ldots,\widehat{x_j},\ldots,x_n) {}$

${} (y'_1,y'_2,\ldots,y'_{n-1}) = (y_1,y_2,\ldots,\widehat{y_i},\ldots,y_n) {}$

${} \hat{C}_{ji} = \prod_{1 \leq k',l' \leq n-1}\frac{1}{x'_{k'}+y'_{l'}} \prod_{1 \leq k' < l' \leq n-1} (x'_{l'}-x'_{k'})(y'_{l'}-y'_{k'}) = \prod_{\substack{1 \leq k,l \leq n \\ k \neq j, l \neq i}}\frac{1}{x_k+y_l} \prod_{\substack{1 \leq k < l \leq n \\ k \neq j \\ l \neq j}} (x_l-x_k) \prod_{\substack{1 \leq k < l \leq n \\ k \neq i \\ l \neq i}} (y_l-y_k) {}$

${} (C^{-1})_{ij} = (-1)^{i+j} \prod_{k=1}^n(x_k + y_i)(x_j+y_k) \cdot \frac{1}{x_j+y_i} \cdot \prod_{\substack{1 \leq k = j < l \leq n}} \frac{1}{x_l-x_j} \cdot \prod_{\substack{1 \leq k < l = j \leq n}} \frac{1}{x_j-x_k} \cdot \prod_{\substack{1 \leq k = i < l \leq n}} \frac{1}{y_l-y_i} \cdot \prod_{\substack{1 \leq k < l = i \leq n}} \frac{1}{y_i-y_k} = {}$

${} = (-1)^{i+j} \prod_{k=1}^n(x_k + y_i)(x_j+y_k) \cdot \frac{1}{x_j+y_i} \cdot \prod_{\substack{j < r \leq n}} (-\frac{1}{x_j-x_r}) \cdot \prod_{\substack{1 \leq r < j}} \frac{1}{x_j-x_r} \cdot \prod_{\substack{i < r \leq n}} (-\frac{1}{y_i-y_r}) \cdot \prod_{\substack{1 \leq r < i \leq n}} \frac{1}{y_i-y_r} = {}$

${} = (-1)^{i+j} \prod_{k=1}^n(x_k + y_i)(x_j+y_k) \cdot \frac{1}{x_j+y_i} \cdot (-1)^{n-j} \prod_{\substack{1 \leq r \leq n\\ r \neq j}} \frac{1}{x_j-x_r} \cdot (-1)^{n-i} \prod_{\substack{1 \leq r \leq n\\ r \neq i}} \frac{1}{y_i-y_r} = {}$

${} = \prod_{r=1}^n(x_r + y_i)(x_j+y_r) \cdot \frac{1}{x_j+y_i} \prod_{\substack{1 \leq r \leq n\\ r \neq j}} \frac{1}{x_j-x_r} \prod_{\substack{1 \leq r \leq n\\ r \neq i}} \frac{1}{y_i-y_r} {}$
