$S = \sum_{k=1}^{n} \frac{1}{2k-1}$

Let $v_p(m) := \max \{ k \in \mathbb{N}_0: p^k \mid m \}, m \in \mathbb{Z} \implies p^{v_p(m)} \mid m, p^{v_p(m) + 1} \not\mid m$

$\forall r\in\mathbb{Q} \exists! x\in\mathbb{Z}, y\in\mathbb{N}: r = \frac{x}{y}, (x,y)=1$

$a(r) := x, b(r) := y$

$v_p(r) := v_p(a(r))$

We need to find $v_2(S)$.

We'll prove $v_2(S) = 2 v_2(n)$

\section*{\$\{\} v\_2(n) = 0 \{\}\$}

$2 \not\mid n$

$D := \prod_{j=1}^n (2j-1)$

$S = \frac{\sum_{k=1}^n \frac{D}{2k-1}}{d} = \frac{a(S)}{b(S)}$

$C := \sum_{k=1}^n \frac{D}{2k-1}$

$S = \frac{C}{D} = \frac{a(S)}{b(S)}\ (*)$

Let $C = ea(S), D = eb(S)$

$2 \not\mid D \implies 2 \not\mid e \implies v_2(C) = v_2(ea(S)) = v_2(e) + v_2(a(S)) = v_2(a(S))$

$v_2(S) = v_2(C)$

$C \equiv \sum_{k=1}^n 1 = n \equiv 1 \pmod 2 \implies v_2(C) = 0$

$v_2(S) = 0 = 2 v_2(n)$

\section*{\$\{\} v\_2(n) > 0 \{\}\$}

$v_2(n) = f > 0$

$n = 2^f g, 2 \not\mid g$

$S = \sum_{k=1}^{\frac{n}{2}}(\frac{1}{2k-1} + \frac{1}{2n-(2k-1)}) = \sum_{k=1}^{\frac{n}{2}}\frac{2n}{(2k-1)(2n-2k+1)} = 2n \overbrace{\sum_{k=1}^{\frac{n}{2}}\frac{1}{(2k-1)(2n-2k+1)}}^{T} = 2nT$

$v_2(S) = v_2(2nT) = v_2(2) + v_2(n) + v_2(T) = f + 1 + v_2(T)$

We need to prove $v_2(T) = f - 1$

$I := \prod_{k=1}^{\frac{n}{2}} (2k-1)(2n-2k+1)$

$H := \sum_{k=1}^{\frac{n}{2}}\frac{I}{(2k-1)(2n-2k+1)}$

$T = \frac{H}{I}$

$2 \not\mid I \implies$ similarly to $(*): v_2(T) = v_2(H)$

$n=2^fg, \frac{n}{2} = 2^{f-1}g, 2 \not\mid g$

$H = \sum_{i=0}^{g-1}\sum_{j=1}^{2^{f-1}} \frac{I}{(2(2^{f-1} i + j) - 1)(2n-2(2^{f-1} i + j) + 1)} = \sum_{i=0}^{g-1}\overbrace{\sum_{j=1}^{2^{f-1}} \underbrace{\frac{I}{(2^{f} i + 2j - 1)(2^{f+1}g-2^{f} i + 2j + 1)}}_{h_{ij}}}^{H_i} = \sum_{i=0}^{g-1} H_i$

$\forall i, 0 \leq i \leq g - 1 \forall j, 1 \leq j \leq 2^{f-1}: h_{ij} \equiv I (2j-1)^{-1} (-(2j-1)^{-1}) = -I (2j-1)^{-2} \pmod {2^f}$

$\{ (2j-1)^{-1} \pmod {2^f}: 1 \leq j \leq 2^{f-1} \} = \{ 1, 3, 5, \ldots, 2^f - 1 \}$

$\{ h_ij \pmod{2^f}: 1 \leq j \leq 2^{f-1} \} = \{ -I\cdot 1^2 \pmod{2^f}, -I\cdot 3^2 \pmod{2^f}, \ldots, -I (2^f-1)^2 \pmod{2^f} \}$

$H_i = \sum_{j=1}^{2^{f-1}} h_{ij} \equiv -I \sum_{j=1}^{2^{f-1}} (2j-1)^2 \pmod{2^f}$

$1^2 + 3^2 + \cdots + (2m-1)^2 = 1^2 + 2^2 + 3^2 + 4^2 + \cdots (2m)^2 - (2^2 + 4^2 + \cdots + (2m)^2) =$

$= 1^2 + 2^2 + 3^2 + 4^2 + \cdots (2m)^2 - 2^2 (1^2 + 2^2 + \cdots + m^2) = \sum_{i=1}^{2m} i^2 - 4 \sum_{i=1}^m i^2 = \frac{2m(2m+1)(4m+1)}{6} - \frac{4m(m+1)(2m+1)}{6} =$

$= \frac{2m(2m+1)}{6}(4m+1 - 2(m+1)) = \frac{2m(2m+1)(2m-1)}{6} = \frac{m(4m^2-1)}{3}$

$H_i \equiv -I \frac{2^{2f-1}(4 \cdot 2^{2f-2}-1)}{3} = -\frac{I}{3} 2^{f-1}(2^{2f} - 1) \equiv \frac{I}{3} 2^{f-1} \pmod{2^f}$

$H = \sum_{i=0}^{g-1} H_i \equiv \frac{I}{3}g 2^{f-1} \pmod{2^f}$

$v_2(H) = v_2(\frac{Ig}{3}) + v_2(2^{f-1}) = f - 1$

$\implies v_2(T) = f-1$

$\implies v_2(S) = 2f$
