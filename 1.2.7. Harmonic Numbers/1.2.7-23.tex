\( \Gamma(x+1) = x \Gamma(x) \)

\( \Gamma'(x+1) = \Gamma(x) + x\Gamma'(x) \)

\( \gamma(x) := \frac{\Gamma'(x)}{\Gamma(x)} \)

\( \gamma(x+1) = \frac{\Gamma(x)}{\Gamma(x+1)} + \frac{x\Gamma'(x)}{\Gamma(x+1)} = \frac{\Gamma(x)}{x\Gamma(x)} + \frac{x\Gamma'(x)}{x\Gamma(x)} = \frac{1}{x} + \gamma(x) \)

\( \gamma(x + 1) - \gamma(x) = \frac{1}{x} \)

\( \gamma(1) = \frac{\Gamma'(1)}{\Gamma(1)} = -\gamma \)

\( \gamma(n) = \gamma(1) + \sum_{k=1}^{n-1} (\gamma(k+1) - \gamma(k)) = -\gamma + \sum_{k=1}^{n-1} \frac{1}{k} = H_{n-1} - \gamma \)

\( H_n = \gamma(n+1) + \gamma \)

\( H(x) := \gamma(x+1) + \gamma \)
