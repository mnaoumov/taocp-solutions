\( A_n := \sum_{k=1}^n\frac{H_k}{n+1-k} \overset{l = n + 1 - k}{=} \sum_{l=1}^n \frac{H_{n+1-l}}{l} = \sum_{l=1}^n \sum_{i=1}^{n+1-l}\frac{1}{il} \overset{s=i+l}{=} \sum_{s=2}^{n+1}\sum_{l=1}^{s-1}\frac{1}{l(s-l)} = \)

\( = \sum_{s=2}^{n+1}\sum_{l=1}^{s-1}(\frac{1}{l} + \frac{1}{s-l})\frac{1}{s} = \sum_{s=2}^{n+1}\frac{1}{s}\sum_{l=1}^{s-1}(\frac{1}{l} + \frac{1}{s-l}) = \sum_{s=2}^{n+1}\frac{2H_{s-1}}{s} = \sum_{s=2}^{n+1}\frac{2H_{s-1}}{s} + \frac{2H_0}{1} = \sum_{s=1}^{n+1}\frac{2H_{s-1}}{s} = \)

\( = 2 \sum_{s=1}^{n+1} \frac{H_s - \frac{1}{s}}{s} = 2 (\sum_{s=1}^{n+1} \frac{H_s}{s} - \sum_{s=1}^{n+1} \frac{1}{s^2}) = 2(\sum_{s=1}^{n+1} \frac{H_s}{s} - H_{n+1}^{(2)}) \)

\( B_n := \sum_{k=1}^n \frac{H_k}{k} = \sum_{k=1}^{n}\frac{1}{k}\sum_{i=1}^k \frac{1}{i} = \sum_{i=1}^n \frac{1}{i}\sum_{k=i}^n \frac{1}{k} = \sum_{i=1}^n \frac{1}{i}(\sum_{k=1}^n \frac{1}{k} - \sum_{k=1}^i \frac{1}{k} + \frac{1}{i}) = \)

\( = \sum_{i=1}^n \frac{1}{i}(H_n - H_i + \frac{1}{i}) = H_n \sum_{i=1}^n \frac{1}{i} - \sum_{i=1}^n \frac{H_i}{i} + \sum_{i=1}^n \frac{1}{i^2} = (H_n)^2 - B_n + H_n^{(2)} \)

\( B_n = \frac{1}{2}((H_n)^2 + H_n^{(2)}) \)

\( A_n = 2(B_{n+1} - H_{n+1}^{(2)}) = (H_{n+1})^2 + H_{n+1}^{(2)} - 2H_{n+1}^{(2)} = (H_{n+1})^2 - H_{n+1}^{(2)} \)
