\( S = \sum_{k=1}^{n} \frac{1}{2k-1} \)

Let \( v_{p}(m) := \max \{ k \in \mathbb{N}_{0}: p^{k} \mid m \}, m \in \mathbb{Z} \implies p^{v_{p}(m)} \mid m, p^{v_{p}(m) + 1} \nmid m \)

\( \forall r\in\mathbb{Q} \exists! x\in\mathbb{Z}, y\in\mathbb{N}: r = \frac{x}{y}, (x,y)=1 \)

\( a(r) := x, b(r) := y \)

\( v_{p}(r) := v_{p}(a(r)) \)

We need to find \( v_{2}(S) \).

We'll prove \( v_{2}(S) = 2 v_{2}(n) \)

\section*{\( v_{2}(n) = 0 \)}

\( 2 \nmid n \)

\( D := \prod_{j=1}^{n} (2j-1) \)

\( S = \frac{\sum_{k=1}^{n} \frac{D}{2k-1}}{d} = \frac{a(S)}{b(S)} \)

\( C := \sum_{k=1}^{n} \frac{D}{2k-1} \)

\( S = \frac{C}{D} = \frac{a(S)}{b(S)}\ (*) \)

Let \( C = ea(S), D = eb(S) \)

\( 2 \nmid D \implies 2 \nmid e \implies v_{2}(C) = v_{2}(ea(S)) = v_{2}(e) + v_{2}(a(S)) = v_{2}(a(S)) \)

\( v_{2}(S) = v_{2}(C) \)

\( C \equiv \sum_{k=1}^{n} 1 = n \equiv 1 \pmod 2 \implies v_{2}(C) = 0 \)

\( v_{2}(S) = 0 = 2 v_{2}(n) \)

\section*{\( v_{2}(n) > 0 \)}

\( v_{2}(n) = f > 0 \)

\( n = 2^{f} g, 2 \nmid g \)

\( S = \sum_{k=1}^{\frac{n}{2}}(\frac{1}{2k-1} + \frac{1}{2n-(2k-1)}) = \sum_{k=1}^{\frac{n}{2}}\frac{2n}{(2k-1)(2n-2k+1)} = 2n \overbrace{\sum_{k=1}^{\frac{n}{2}}\frac{1}{(2k-1)(2n-2k+1)}}^{T} = 2nT \)

\( v_{2}(S) = v_{2}(2nT) = v_{2}(2) + v_{2}(n) + v_{2}(T) = f + 1 + v_{2}(T) \)

We need to prove \( v_{2}(T) = f - 1 \)

\( I := \prod_{k=1}^{\frac{n}{2}} (2k-1)(2n-2k+1) \)

\( H := \sum_{k=1}^{\frac{n}{2}}\frac{I}{(2k-1)(2n-2k+1)} \)

\( T = \frac{H}{I} \)

\( 2 \nmid I \implies \) similarly to \( (*): v_{2}(T) = v_{2}(H) \)

\( n=2^{f}g, \frac{n}{2} = 2^{f-1}g, 2 \nmid g \)

\( H = \sum_{i=0}^{g-1}\sum_{j=1}^{2^{f-1}} \frac{I}{(2(2^{f-1} i + j) - 1)(2n-2(2^{f-1} i + j) + 1)} = \sum_{i=0}^{g-1}\overbrace{\sum_{j=1}^{2^{f-1}} \underbrace{\frac{I}{(2^{f} i + 2j - 1)(2^{f+1}g-2^{f} i + 2j + 1)}}_{h_{ij}}}^{H_{i}} = \sum_{i=0}^{g-1} H_{i} \)

\( \forall i, 0 \leq i \leq g - 1 \forall j, 1 \leq j \leq 2^{f-1}: h_{ij} \equiv I {(2j-1)}^{-1} (-{(2j-1)}^{-1}) = -I {(2j-1)}^{-2} \pmod {2^{f}} \)

\( \{ {(2j-1)}^{-1} \pmod {2^{f}}: 1 \leq j \leq 2^{f-1} \} = \{ 1, 3, 5, \ldots, 2^{f} - 1 \} \)

\( \{ h_{i}j \pmod{2^{f}}: 1 \leq j \leq 2^{f-1} \} = \{ -I\cdot 1^{2} \pmod{2^{f}}, -I\cdot 3^{2} \pmod{2^{f}}, \ldots, -I {(2^{f}-1)}^{2} \pmod{2^{f}} \} \)

\( H_{i} = \sum_{j=1}^{2^{f-1}} h_{ij} \equiv -I \sum_{j=1}^{2^{f-1}} {(2j-1)}^{2} \pmod{2^{f}} \)

\( 1^{2} + 3^{2} + \cdots + {(2m-1)}^{2} = 1^{2} + 2^{2} + 3^{2} + 4^{2} + \cdots {(2m)}^{2} - (2^{2} + 4^{2} + \cdots + {(2m)}^{2}) = \)

\( = 1^{2} + 2^{2} + 3^{2} + 4^{2} + \cdots {(2m)}^{2} - 2^{2} (1^{2} + 2^{2} + \cdots + m^{2}) = \sum_{i=1}^{2m} i^{2} - 4 \sum_{i=1}^{m} i^{2} = \frac{2m(2m+1)(4m+1)}{6} - \frac{4m(m+1)(2m+1)}{6} = \)

\( = \frac{2m(2m+1)}{6}(4m+1 - 2(m+1)) = \frac{2m(2m+1)(2m-1)}{6} = \frac{m(4m^{2}-1)}{3} \)

\( H_{i} \equiv -I \frac{2^{2f-1}(4 \cdot 2^{2f-2}-1)}{3} = -\frac{I}{3} 2^{f-1}(2^{2f} - 1) \equiv \frac{I}{3} 2^{f-1} \pmod{2^{f}} \)

\( H = \sum_{i=0}^{g-1} H_{i} \equiv \frac{I}{3}g 2^{f-1} \pmod{2^{f}} \)

\( v_{2}(H) = v_{2}(\frac{Ig}{3}) + v_{2}(2^{f-1}) = f - 1 \)

\( \implies v_{2}(T) = f-1 \)

\( \implies v_{2}(S) = 2f \)
