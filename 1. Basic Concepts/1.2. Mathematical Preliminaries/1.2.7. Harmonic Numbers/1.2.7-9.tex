\( F(x) = \sum_{k=1}^{n} \binom{n}{k} x^{k}H_{k} \)

\begin{multline*}
F(-1) = \sum_{k=1}^{n} \binom{n}{k} {(-1)}^{k} H_{k} = \sum_{k=1}^{n} \binom{n}{k} {(-1)}^{k} \sum_{j=1}^{k} \frac{1}{j} \\
= \sum_{j=1}^{n} \frac{1}{j} \sum_{k=j}^{n} \binom{n}{k} {(-1)}^{k}
\end{multline*}

\begin{multline*}
\sum_{k=j}^{n} \binom{n}{k} {(-1)}^{k} = \sum_{k=1}^{n} \binom{n}{k} {(-1)}^{k} - \sum_{k=0}^{j-1} \binom{n}{k} {(-1)}^{k} \\
= 0 - \sum_{k=0}^{j-1} \binom{n}{k} {(-1)}^{k} \overset{1.2.6(18)}{=}
\end{multline*}

\( = -{(-1)}^{j-1} \binom{n-1}{j-1} = {(-1)}^{j} \binom{n-1}{j-1} \)

\begin{multline*}
F(-1) = \sum_{j=1}^{n} \frac{1}{j} {(-1)}^{j} \binom{n-1}{j-1} = \sum_{j=1}^{n} \frac{1}{n} {(-1)}^{j} \binom{n}{j} \\
= \frac{1}{n} (\sum_{j=0}^{n} {(-1)}^{j} \binom{n}{j} - {(-1)}^{0}\binom{n}{0}) = -\frac{1}{n}
\end{multline*}
