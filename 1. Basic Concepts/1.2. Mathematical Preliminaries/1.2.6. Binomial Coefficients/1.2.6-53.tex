\subsubsection*{a)}

\( \sum_{k=0}^m \binom{r}{k}\binom{s}{n-k}(nr-(r+s)k) = (m+1)(n-m) \binom{r}{m+1}\binom{s}{n-m} \)

\( m = 0 \)

\( \binom{r}{0}\binom{s}{0}(nr-(r+s)0) = (0+1)(n-0) \binom{r}{0+1}\binom{s}{n-0} \)

\( (m \implies m + 1) \)

\( \sum_{k=0}^{m+1} \binom{r}{k}\binom{s}{n-k}(nr-(r+s)k) = (m+1)(n-m) \binom{r}{m+1}\binom{s}{n-m} + \binom{r}{m+1}\binom{s}{n-m-1}(nr - (r+s)(m+1)) = \)

\( = \binom{r}{m+1}\binom{s}{n-m-1} ((m+1)(n-m) \cdot \frac{s+1-n+m}{n-m} + nr - (r+s)(m+1)) = \)

\( = \binom{r}{m+1}\binom{s}{n-m-1} ((m+1)(s+1-n+m) + nr - (r+s)(m+1)) = \)

\( = \binom{r}{m+1}\binom{s}{n-m-1} ((m+1)(1-n+m-r) + nr) =: A \)

\( (m+1)(1-n+m-r) + nr = r(n-(m+1)) + r(m+1) + (m+1)(1-n+m-r) = \)

\( r(n-m-1) + (m+1)(1-n+m) = (r-m-1)(n-m-1) \)

\( A = \binom{r}{m+1}\binom{s}{n-m-1} (r-m-1)(n-m-1) = \binom{r}{m+2} \frac{m+2}{r-m-1} \binom{s}{n-m-1} (r-m-1)(n-m-1) = \)

\( = ((m+1)+1)(n-(m+1)) \binom{r}{(m+1)+1} \binom{s}{n-(m+1)} \)

\subsubsection*{b)}

\( \binom{-\frac{1}{2}}{n} = (-\frac{1}{4})^n \binom{2n}{n} \)

\( \binom{\frac{1}{2}}{n} = \frac{\Gamma(\frac{3}2)}{\Gamma(n+1)\Gamma(\frac{1}{2}-n+1)} = \frac{\Gamma(\frac{3}{2})}{\Gamma(n+1)\Gamma(\frac{3}{2}-n)} = \frac{\frac{1}{2}\Gamma(\frac{1}{2})}{\Gamma(n+1)\Gamma(\frac{1}{2}-n) (\frac{1}{2}-n)} = \frac{1}{1-2n} \binom{-\frac{1}{2}}{n} = \frac{1}{1-2n} \cdot (-\frac{1}{4})^n \binom{2n}{n} \)

\( n > 0 \implies \binom{2n}{n} = \binom{2n-1}{n}\cdot\frac{2n}{2n-n} = 2 \binom{2n-1}{n} \)

\( n = 0: \binom{2n}{n} = 1, 2 \binom{2n-1}{n} = 2 \)

\( \binom{2n}{n} = 2 \binom{2n-1}{n} - \delta_{n0} \)

\( \binom{\frac{1}{2}}{n} = \frac{(-1)^{n-1}}{(2n-1) 2^{2n-1}} \binom{2n-1}{n} - \delta_{n0} \)

\( r = \frac{1}{2}, s = -\frac{1}{2} \)

\( \sum_{k=0}^m \binom{\frac{1}{2}}{k}\binom{-\frac{1}{2}}{n-k} \cdot nr = (m+1)(n-m) \binom{\frac{1}{2}}{m+1}\binom{-\frac{1}{2}}{n-m} \)

\( \sum_{k=0}^m \binom{\frac{1}{2}}{k}\binom{-\frac{1}{2}}{n-k} \cdot \frac{n}{2} = (m+1)(n-m) \binom{\frac{1}{2}}{m+1}\binom{-\frac{1}{2}}{n-m} \)

\( \sum_{k=0}^m \Bigl(\frac{(-1)^{k-1}}{(2k-1) 2^{2k-1}} \binom{2k-1}{k} - \delta_{k0}\Bigr) \frac{(-1)^{n-k}}{2^{2n-2k}} \binom{2n-2k}{n-k} \cdot \frac{n}{2} = (m+1)(n-m) \Bigl(\frac{(-1)^m}{(2m+1) 2^{2m+1}} \binom{2m+1}{m+1} - \delta_{m+1,0}\Bigr) \frac{(-1)^{n-m}}{2^{2n-2m}} \binom{2n-2m}{n-m} \)

\( \Bigl(\frac{(-1)^{n-1}}{2^{2n-1}}\sum_{k=0}^m \frac{1}{2k-1} \binom{2k-1}{k} \binom{2n-2k}{n-k} - \frac{(-1)^{n}}{2^{2n}} \binom{2n}{n}\Bigr) \cdot \frac{n}{2} = (m+1)(n-m) \frac{(-1)^n}{(2m+1) 2^{2n+1}} \binom{2m+1}{m+1} \binom{2n-2m}{n-m} \)

\( \frac{(-1)^{n-1}}{2^{2n-1}} (\sum_{k=0}^m \frac{1}{2k-1} \binom{2k-1}{k} \binom{2n-2k}{n-k} + \frac{1}{2} \binom{2n}{n}) \cdot \frac{n}{2} = (m+1)(n-m) \frac{(-1)^n}{(2m+1) 2^{2n+1}} \binom{2m+1}{m+1} \binom{2n-2m}{n-m} \)

\( \sum_{k=0}^m \frac{1}{2k-1} \binom{2k-1}{k} \binom{2n-2k}{n-k} = - \frac{1}{2} \binom{2n}{n} - \frac{1}{2n} (m+1)(n-m) \frac{1}{2m+1} \binom{2m+1}{m+1} \binom{2n-2m}{n-m} \)

\( \sum_{k=0}^m \binom{-1-(-2)k}{k} \binom{0 - (-2)(n-k)}{n-k} \frac{-1}{-1-(-2)k} = \frac{1}{2} \binom{2n}{n} + \frac{1}{2n} (m+1)(n-m) \frac{1}{2m+1} \binom{2m+1}{m+1} \binom{2n-2m}{n-m} = \)

\( =  \frac{1}{2} \binom{2n}{n} + \frac{1}{2n} (n-m) \binom{2m}{m} \binom{2n-2m}{n-m} \)
