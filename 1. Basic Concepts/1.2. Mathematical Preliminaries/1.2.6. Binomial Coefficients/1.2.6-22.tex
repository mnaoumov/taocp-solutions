\( (26): \sum_{k} \binom{r-tk}{k}\binom{s-t(n-k)}{n-k}\frac{r}{r-tk} = \binom{r+s-tn}{n} \)

\( s = n - 1 - r + tn \)

\( RHS = \binom{r+n-1-r+tn-tn}{n} = \binom{n-1}{n} = \delta_{n0} \)

\( LHS = \sum_k \binom{r-tk}{k}\binom{n-1-r + tn - tn + tk}{n-k}\frac{r}{r-tk} = \sum_k \binom{r-tk}{k}\binom{n - 1 - r + tk}{n-k}\frac{r}{r-tk} \)

\subsubsection*{1. \$\{\} n < 0 \{\}\$}

\( LHS=0=RHS \)

\subsubsection*{2. \$\{\} n = 0 \{\}\$}

\( LHS = \binom{r-t\cdot 0}{0}\binom{n-1-r+t\cdot 0}{0}\frac{r}{r-t\cdot 0} = 1 = RHS \)

\subsubsection*{3. \$\{\} n > 0 \{\}\$}

\( LHS = \sum_k \binom{r-tk}{k}\binom{n - 1 - r + tk}{n-k}\frac{r}{r-tk} \)

\( \binom{r-tk}{k}\binom{n - 1 - r + tk}{n-k}\frac{r}{r-tk} = \frac{\prod_{j=0}^{k-1}(r-tk-j)}{k!} \cdot \frac{\prod_{j=0}^{n-k-1} (n-1-r+tk-j)}{(n-k)!} \cdot \frac{r}{r-tk} = \frac{r}{n!} \binom{n}{k} (-1)^{k-1} \prod_{j=1}^{k-1} (-r+tk+j) \cdot \prod_{j=0}^{n-k-1} (n-1-r+tk-j) = (-1)^{n-1} \binom{n}{k} (-1)^{n-k} P_{n-1}(k) \)

\( P_{n-1}(x) = \prod_{j=1}^{k-1}(tx - r + j) \cdot \prod_{j=0}^{n-k-1}(tx - r + (n - 1 - j)) = \prod_{j=1}^{n - 1}(tx-r + j) \)

\( LHS = (-1)^{n-1}\sum_{k} \sum_k \binom{n}{k}(-1)^{n-k} P_{n-1}(k) \overset{(34)}{=} 0 = RHS \)
