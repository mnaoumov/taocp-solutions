\( n = pq + r \)

\begin{multline*}
\binom{n}{p} = \frac{(pq+r)(pq+r-1)\cdots(pq+1)pq(pq-1)\cdots(pq-(p-r-1))}{p!} \\
= \langle\frac{q(pq+r)(pq+r-1)\cdots(pq+1)(pq-1)\cdots(pq-(p-r-1))}{(p-1)!}\rangle_{p} =
\end{multline*}

\[ = \langle q \rangle_{p} \cdot \langle(p-1)!\rangle_{p} \cdot \langle{((p-1)!)}^{-1}\rangle_{p} = \langle q \rangle_{p} = \langle \lfloor\frac{n}{p}\rfloor \rangle_{p} \]

\subsubsection*{(b)}

\begin{multline*}
\binom{p}{k} = \frac{p(p-1)(p-2)\cdots(p-k+1)}{k!} \\
= p\langle(p-1)(p-2)\cdots(p-k+1)\rangle_{p} \cdot \langle{(k!)}^{-1}\rangle_{p} = \langle 0 \rangle_{p}
\end{multline*}

\subsubsection*{(c)}

\[ \binom{p-1}{k} = \frac{(p-1)(p-2)\cdots(p-k)}{k!} = \langle{(-1)}^{k} k!\rangle_{p} \cdot \langle{(k!)}^{-1}\rangle_{p} = \langle{(-1)}^{k} \rangle_{p} \]

\subsubsection*{(d)}

\begin{multline*}
\binom{p+1}{k} = \frac{(p+1)p(p-1)\cdots(p-k+2)}{k!} \\
= p\langle{(-1)}^{k-2}(k-2)!\rangle_{p} \cdot \langle{(k!)}^{-1}\rangle_{p} = \langle 0 \rangle_{p}
\end{multline*}

\subsubsection*{(e)}

\( n = pq + r \)

\( k = ps + t \)

\( k! = (ps+t)! = \)

\( = 1 \cdot 2 \cdot \cdots \cdot (p-1) \cdot p \cdot \)

\( \cdot (p+1) \cdot (p+2) \cdot \cdots \cdot (p + p-1) \cdot 2p \cdot \)

\( \cdots \)

\( \cdot (p(s-1)+1) \cdot (p(s-1)+2) \cdot \cdots \cdot (p(s-1) + p-1) \cdot ps \cdot \)

\( \cdot (ps+1) \cdot (ps+2) \cdot \cdots \cdot (ps + t) = \)

\[ = {(\langle(p-1)!\rangle_{p})}^{s} \cdot p^{s} s! \cdot \langle t!\rangle_{p} = {(\langle-1\rangle_{p})}^{s} \cdot p^{s} s! \cdot \langle t!\rangle_{p} \]

Similarly, \( n! = {(\langle-1\rangle_{p})}^{q} \cdot p^{q} q! \cdot \langle r!\rangle_{p} \)

\( n - k = p(q-s) + (r - t) \)

\paragraph*{1. \$\{\} r \mbox{-} t \textbackslash\{\}geq 0 \{\}\$}

\( (n-k)! = {(\langle-1\rangle_{p})}^{q-s} \cdot p^{q-s} (q-s)! \cdot \langle (r-t)!\rangle_{p} \)

\begin{multline*}
\binom{n}{k} = \frac{n!}{k!(n-k)!} \\
= \frac{{(\langle-1\rangle_{p})}^{q} \cdot p^{q} q! \cdot \langle r!\rangle_{p}}{{(\langle-1\rangle_{p})}^{s} \cdot p^{s} s! \cdot \langle t!\rangle_{p} \cdot {(\langle-1\rangle_{p})}^{q-s} \cdot p^{q-s} (q-s)! \cdot \langle (r-t)!\rangle_{p}} \\
= \langle\binom{q}{s}\binom{r}{t}\rangle_{p} = \langle\binom{\lfloor\frac{n}{p}\rfloor}{\lfloor\frac{k}{p}\rfloor}\binom{n \mod p}{k \mod p}\rangle_{p}
\end{multline*}

\paragraph*{2. \$\{\} r \mbox{-} t < 0 \{\}\$}

\( \binom{r}{t} = 0 \)

\( n - k = p(q-s-1) + (p + r -t) \)

\begin{multline*}
\binom{n}{k} = \frac{n!}{k!(n-k)!} \\
= \frac{{(\langle-1\rangle_{p})}^{q} \cdot p^{q} q! \cdot \langle r!\rangle_{p}}{{(\langle-1\rangle_{p})}^{s} \cdot p^{s} s! \cdot \langle t!\rangle_{p} \cdot {(\langle-1\rangle_{p})}^{q-s-1} \cdot p^{q-s-1} (q-s-1)! \cdot \langle (p+r-t)!\rangle_{p}} \\
= \langle 0 \rangle_{p} = \langle\binom{\lfloor\frac{n}{p}\rfloor}{\lfloor\frac{k}{p}\rfloor}\binom{n \mod p}{k \mod p}\rangle_{p}
\end{multline*}
