\( {(x+y)}^{\underline{n}} = \sum_{k} \binom{n}{k} x^{\underline{k}}y^{\underline{n-k}} \)

\( n = 0 \)

\( {(x+y)}^{\underline{0}} = 1 = \binom{0}{0} x^{\underline{0}}y^{\underline{0}} \)

\( (n \implies n + 1) \)

\( {(x+y)}^{\underline{n+1}} = (x+y){(x+y-1)}^{\underline{n}} = x{((x-1)+y)}^{\underline{n}} + y{(x+(y-1))}^{\underline{n}} = x \sum_{k} \binom{n}{k} {(x-1)}^{\underline{k}}y^{\underline{n-k}} + y \sum_{k} \binom{n}{k} x^{\underline{k}}{(y-1)}^{\underline{n-k}} = \)

\( = \sum_{k} \binom{n}{k} x^{\underline{k+1}}y^{\underline{n-k}} + \sum_{k} \binom{n}{k} x^{\underline{k}}y^{\underline{n+1-k}} = \sum_{k} \binom{n}{k-1} x^{\underline{k}}y^{\underline{n+1-k}} + \sum_{k} \binom{n}{k} x^{\underline{k}}y^{\underline{n+1-k}} = \sum_{k} (\binom{n}{k-1} + \binom{n}{k})x^{\underline{k}}y^{\underline{n+1-k}} = \)

\( = \sum_{k} \binom{n+1}{k}x^{\underline{k}}y^{\underline{n+1-k}} \)

\bigskip\hrule\bigskip

\( {(x+y)}^{\overline{n}} = {(-1)}^{n} {(-x-y)}^{\underline{n}} = {(-1)}^{n} \sum_{k} \binom{n}{k} {(-x)}^{\underline{k}}{(-y)}^{\underline{n-k}} = \sum_{k} \binom{n}{k} {(-1)}^{k}{(-x)}^{\underline{k}}{(-1)}^{n-k}{(-y)}^{\underline{n-k}} = \)

\( = \sum_{k} \binom{n}{k}x^{\overline{k}}y^{\overline{n-k}} \)
