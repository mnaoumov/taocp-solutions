\( n! = \lim_{m\to\infty} \frac{m^{n}m!}{(n+1)(n+2) \cdot\ \cdots\ \cdot (n+m)}\quad (13) \)

\begin{multline*}
\frac{m^{n}m!}{(n+1)(n+2) \cdot\ \cdots\ \cdot (n+m)} = \frac{m^{n} m! n!}{(n+m)!} = \\
\frac{m^{n} n!}{(m+1)(m+2) \cdot\ \cdots\ \cdot (m+n)} = \\
\frac{n!}{(1+\frac{1}{m})(1+\frac{2}{m}) \cdot\ \cdots\ \cdot (1+\frac{n}{m})} \to n!, m \to \infty
\end{multline*}
