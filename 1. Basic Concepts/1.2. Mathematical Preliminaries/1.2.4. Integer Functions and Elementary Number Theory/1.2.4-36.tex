\( \sum_{k=1}^{n} \lfloor \frac{k}{2} \rfloor = \sum_{j=1}^{\lfloor\frac{n}{2}\rfloor} j \sum_{k:1 \leq k \leq n, \lfloor \frac{k}{2} \rfloor = j} 1 = \sum_{j=1}^{\lfloor\frac{n}{2}\rfloor} 2j - \lfloor\frac{n}{2} \rfloor [2 \backslash n] =: A \)

\( n = 2m \implies A = m(m+1) - m = m^{2} = \frac{n^{2}}{4} = \lfloor \frac{n^{2}}{4} \rfloor \)

\( n = 2m + 1 \implies A = m(m+1) = m^{2} + m = \frac{4m^{2}+4m}{4} = \frac{{(2m+1)}^{2}-1}{4} = \frac{n^{2}-1}{4} = \lfloor \frac{n^{2}-1}{4} + \frac{1}{4} \rfloor =  \lfloor \frac{n^{2}}{4} \rfloor \)

\( \sum_{k=1}^{n} \lceil \frac{k}{2} \rceil = \sum_{j=1}^{\lceil\frac{n}{2}\rceil} j \sum_{k:1 \leq k \leq n, \lceil \frac{k}{2} \rceil = j} 1 = -1 + \sum_{j=1}^{\lceil\frac{n}{2}\rceil} 2j - \lceil\frac{n}{2} \rceil [\lnot(2 \backslash n)] =: B \)

\( n = 2m \implies B = -1 + m(m+1) = m^{2}+m-1 = \frac{4m^{2}+4m-4}{4} = \frac{n^{2}+2n-4}{4} = \frac{{(n+1)}^{2}}{4} - 1 + \frac{1}{4} = \lceil \frac{{(n+1)}^{2}}{4} \rceil - 1 \)

\( n = 2m-1 \implies B = -1 + m(m+1) - m = m^{2} - 1 = \frac{{(n+1)}^{2}}{4} - 1 = \lceil \frac{{(n+1)}^{2}}{4} \rceil - 1 \)
