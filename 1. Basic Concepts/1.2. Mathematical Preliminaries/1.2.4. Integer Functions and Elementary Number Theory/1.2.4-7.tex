\( \lfloor x \rfloor + \lfloor y \rfloor \leq x + y \implies \lfloor x \rfloor + \lfloor y \rfloor \leq \lfloor x + y \rfloor \)

\( n := \lfloor x \rfloor, m := \lfloor y \rfloor \)

\( \alpha := x \mod 1, \beta := y \mod 1 \)

\( x + y = n + m + \alpha + \beta \)

\( \lfloor x \rfloor + \lfloor y \rfloor = \lfloor x + y \rfloor \iff \alpha + \beta < 1 \)

\( \lceil x \rceil + \lceil y \rceil \geq x + y \implies \lceil x \rceil + \lceil y \rceil \geq \lceil x + y \rceil \)

\( n := \lceil x \rceil, m := \lceil y \rceil \)

\( \alpha := (-x) \mod 1, \beta := (-y) \mod 1 \)

\( x = n - \alpha, y = m - \beta \)

\( x + y = n + m - (\alpha + \beta) \)

\( \lceil x \rceil + \lceil y \rceil = x + y \iff \alpha + \beta < 1 \)
