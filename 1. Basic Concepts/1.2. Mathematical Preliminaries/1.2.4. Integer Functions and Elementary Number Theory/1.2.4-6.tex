\subsubsection*{(a)}

\( n = \lfloor \sqrt{\lfloor x\rfloor} \rfloor \iff n \leq \sqrt{\lfloor x\rfloor} < n + 1 \iff n^{2} \leq \lfloor x\rfloor < {(n+1)}^{2} \iff n^{2} \leq x < {(n+1)}^{2} \iff n \leq \sqrt{x} < n + 1 \iff n = \lfloor \sqrt{x} \rfloor \)

\subsubsection*{(b)}

\( n = \lceil \sqrt{\lceil x\rceil} \rceil \iff n - 1 < \sqrt{\lceil x\rceil} \leq n \iff {(n-1)}^{2} < \lceil x\rceil \leq n^{2} \iff {(n-1)}^{2} < x \leq n^{2} \iff n - 1 < \sqrt{x} \leq n \iff n = \lceil \sqrt{x} \rceil \)
