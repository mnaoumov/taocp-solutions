\subsubsection*{f)} % chktex 9 % chktex 10

\( f(x) = \log |2 \sin \pi x | \)

\( \sum_{k=0}^{n-1} f(x+\frac{k}{n}) = \sum_{k=0}^{n-1} \log |2 \sin \pi (x + \frac{k}{n}| = \log |\prod_{k=0}^{n-1}(2\sin \pi (x + \frac{k}{n}))| =: A \)

\( \omega_{k} := e^{-\frac{2\pi i k}{n}} \)

\( 2\sin \pi t = \frac{e^{\pi i t}-e^{-\pi i t}}{i} = e^{\pi i (t - \frac{1}{2})}(1-e^{-2\pi i t}) = e^{\pi i (t - \frac{1}{2})}(1-e^{-2\pi i t}) \)

\( A = \overbrace{e^{i \pi \sum_{k=0}^{n-1}(x+\frac{k}{n}-\frac{1}{2})}}^{B} \overbrace{\prod_{k=0}^{n-1} (1-e^{-2\pi i x}\omega_{k})}^{C} \)

a) \( \implies B = e^{i \pi (nx - \frac{1}{2})} \)

\( g(t) := \prod_{k=0}^{n-1} (1-t\omega_{k}) \)

\( \forall k, 0 \leq k \leq n - 1: g(\frac{1}{\omega_{k}}) = 0 \)

\( h(t) = 1-t^{n} \)

\( \forall k, 0 \leq k \leq n - 1: h(\frac{1}{\omega_{k}}) = 0 \)

\( g, h \) have the same coefficient to \( t^{0} \)

\( \implies g(t) = h(t) \)

\( C = g(e^{-2\pi i x}) = 1 - e^{-2\pi i n x} \)

\( \implies A = f(nx) \)

\( \sum_{k=0}^{n-1} g(x+\frac{k}{n}) = \sum_{k=0}^{n-1} f(x+\frac{k}{n} - \lfloor x+\frac{k}{n} \rfloor) =: A \)

\( m = nq + r, 0 \leq r \leq m - 1 \)

\( m \leq nx < m + 1 \implies \frac{m}{n} \leq x < \frac{m+1}{n} \)

\( \frac{m+k}{n} \leq x + \frac{k}{n} < \frac{m+k+1}{n} \)

\( \frac{nq+r+k}{n} \leq x + \frac{k}{n} < \frac{nq+r+k+1}{n} \)

\( q + \lfloor \frac{r+k}{n} \rfloor \leq \lfloor x + \frac{k}{n} \rfloor < q + \lfloor \frac{r+k+1}{n} \rfloor \)
