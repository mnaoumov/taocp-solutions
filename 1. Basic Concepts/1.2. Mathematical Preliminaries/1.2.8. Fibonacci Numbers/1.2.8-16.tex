\( a_{n} := \sum_{k=0}^{n} \binom{n-k}{k} = \sum_{k} \binom{n-k}{k} \)

\( a_{0} = \binom{0}{0} = 1 \)

\( a_{1} = \binom{1}{0} + \binom{0}{1} = 1 \)

\( a_{n+2} = \sum_{k}\binom{n+2-k}{k} = \sum_{k}(\binom{n+1-k}{k} + \binom{n+1-k}{k-1}) = \sum_{k}\binom{n+1-k}{k} + \sum_{k}\binom{n-(k-1)}{k-1} = a_{n+1} + a_{n} \)

\( a_{n} = F_{n+1} \)
