\section{Length}\label{sec:length}

\( S_{1} = \text{``a''}, S_{2} = \text{``b''}, S_{n+2} = S_{n+1}S_{n}, n \geq 1 \)

\( l(S_{1}) = 1 = F_{1} \)

\( l(S_{2}) = 1 = F_{2} \)

\( l(S_{n+2}) = l(S_{n+1}) + l(S_{n}) = F_{n+1} + F_{n} = F_{n+2} \)

\( \implies l(S_{n}) = F_{n} \)

\section{Count of ``a'' and ``b''}\label{sec:count}

Let \( a_{n} - \) amount of \text{``a''} in \( S_{n} \).

\( b_{n} - \) amount of \text{``b''} in \( S_{n} \).

\( a_{1} = 1 = F_{-1} \)

\( a_{2} = 0 = F_{0} \)

\( a_{n+2} = a_{n+1} + a_{n} \)

\( \implies a_{n} = F_{n-2} \)

\( b_{1} = 0 = F_{0} \)

\( b_{2} = 1 = F_{1} \)

\( b_{n+2} = b_{n+1} + b_{n} \)

\( \implies b_{n} = F_{n-1} \)

\section{Density of ``b''}\label{sec:density}

\( db_{n} = \frac{b_{n}}{l(S_{n})} = \frac{F_{n-1}}{F_{n}} \to \phi^{-1}, n \to \infty \)

\section{First letter}\label{sec:first-letter}

For \( n \geq 2: S_{n}[1] = \text{``b''} \).

\( n = 2: S_{2}[1] = \text{``b''}[1] = \text{``b''} \implies \) (\ref{sec:first-letter})

\( \forall n \geq 3: S_{n}[1] = (S_{n-1}S_{n-2})[1] = S_{n-1}[1] = \text{``b''} \implies \) (\ref{sec:first-letter})

\section{Last letter}\label{sec:last-letter}

\( S_{n}[F_{n}] = \text{``a''} \iff 2 \nmid n \).

Consider \( c_{k} := S_{2k-1}[F_{2k-1}], d_{k} := S_{2k}[F_{2k}] \)

Let's prove \( c_{k} = \text{``a''}, d_{k} = \text{``b''} \).

\( c_{1} = S_{1}[1] = \text{``a''}[1] = \text{``a''} \)

\( c_{k+1} = S_{2k+1}[F_{2k+1}] = (S_{2k}S_{2k-1})[F_{2k+1}] = S_{2k-1}[F_{2k-1}] = c_{k} = \text{``a''} \)

\( d_{1} = S_{2}[1] = \text{``b''}[1] = \text{``b''} \)

\( d_{k+1} = S_{2k+2}[F_{2k+2}] = (S_{2k+1}S_{2k})[F_{2k+2}] = S_{2k}[F_{2k}] = d_{k} = \text{``b''} \)

\section{No ``aa''}\label{sec:no-aa}

No ``aa'' appears in \( S_{n} \).

\( n = 1: S_{1} = \text{``a''} \implies \) (\ref{sec:no-aa})

\( n = 2: S_{2} = \text{``b''} \implies \) (\ref{sec:no-aa})

\( n = 3: S_{3} = \text{``ba''} \implies \) (\ref{sec:no-aa})

\( n \geq 4: S_{n} = S_{n-1}S_{n-2} \)

By inductive assumption, \( S_{n-1} \) and \( S_{n-2} \) do not contain ``aa''.

The only possible way to get ``aa'' in \( S_{n} \) is to have ``a'' at the end of \( S_{n-1} \) and ``a'' at the beginning of \( S_{n-2} \).

But \( n - 2 \geq 2 \implies S_{n-2} \) starts with ``b''.

\( \implies \) (\ref{sec:no-aa})

\section{New ``bb''}\label{sec:new-bb}

New ``bb'' appears in \( S_{n} \) for \( n = 3 + 2k, k \geq 1 \).

New ``bb'' appears in \( S_{n} \) if and only if \( S_{n-1} \) ends with ``b'' and \( S_{n-2} \) starts with ``b''.

\( S_{n-1} \) ends with ``b'' \( \iff 2 | (n-1) \)

\( S_{n-2} \) starts with ``b'' \( \iff n - 2 \geq 2 \)

\( \implies \) (\ref{sec:new-bb})

\section{Letter ``a'' at position \( k \)}\label{sec:a-at-k}

\( S_{n}[k] = \text{``a''} \iff k - 1 = F_{m_{1}} + F_{m_{2}} + \cdots + F_{m_{r}}, m_{1} \gg m_{2} \gg \cdots \gg m_{r} = 2 \)

\( S_{n}[k] = S_{m_{r}+1}[F_{m_{r}}+1] = (S_{m_{r}}S_{m_{r}-1})[F_{m_{r}}+1] = S_{m_{r}-1}[1] \)

\( S_{m_{r}-1}[1] = \text{``a''} \iff m_{r}-1 = 1 \iff m_{r} = 2 \)

\section{No starting ``bb''}\label{sec:no-starting-bbb}

No starting ``bb'' appears in \( S_{n} \).

\( n = 1: S_{1} = \text{``a''} \implies \) (\ref{sec:no-starting-bbb})

\( n = 2: S_{2} = \text{``b''} \implies \) (\ref{sec:no-starting-bbb})

\( n = 3: S_{3} = \text{``ba''} \implies \) (\ref{sec:no-starting-bbb})

\( n \geq 4: S_{n}[1..2] = (S_{n-1}S_{n-2})[1..2] = S_{n-1}[1..2] \neq \text{``bb''} \implies \) (\ref{sec:no-starting-bbb})

\section{No ending ``bb''}\label{sec:no-ending-bbb}

No ending ``bb'' appears in \( S_{n} \).

\( n = 1: S_{1} = \text{``a''} \implies \) (\ref{sec:no-ending-bbb})

\( n = 2: S_{2} = \text{``b''} \implies \) (\ref{sec:no-ending-bbb})

\( n = 3: S_{3} = \text{``ba''} \implies \) (\ref{sec:no-ending-bbb})

\( n = 4: S_{4} = \text{``bab''} \implies \) (\ref{sec:no-ending-bbb})

\( n \geq 5: S_{n}[(F_{n}-1)..F_{n}] = (S_{n-1}S_{n-2})[(F_{n}-1)..F_{n}] = S_{n-2}[(F_{n_{2}}-1)..F_{n-2}] \neq \text{``bb''} \implies \) (\ref{sec:no-ending-bbb})

\section{No ``bbb''}\label{sec:no-bbb}

No ``bbb'' appears in \( S_{n} \).

In order to get ``bbb'' you have have to have

either

- ``bb'' at the ending of \( S_{n-1} \) and ``b'' at the beginning of \( S_{n-2} \).

- ``b'' at the ending of \( S_{n-1} \) and ``bb'' at the beginning of \( S_{n-2} \).

By sections~\ref{sec:no-starting-bbb} and~\ref{sec:no-ending-bbb} both cases are impossible.
