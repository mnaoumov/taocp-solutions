\( A(z) := \sum_{n \geq 0}a_{n}z^{n} = z + \sum_{n \geq 0}(a_{n+1} + a_{n} + \binom{n}{m})z^{n+2} = \)

\( = z + z(A(z) - 0) + z^{2} A(z) + \sum_{n \geq 0}\binom{n}{m} z^{n+2} = \)

\( = z + A(z)(z+z^{2}) + \sum_{n \geq -m}\binom{n+m}{m} z^{n+m+2} = \)

\[ = z + A(z)(z+z^{2}) + \sum_{n \geq 0}\binom{n+m}{m} z^{n+m+2} = z+A(z)(z+z^{2})+z^{m+2}\frac{1}{{(1-z)}^{m+1}} \]

\begin{multline*}
A(z) = \frac{z + \frac{z^{m+2}}{{(1-z)}^{m+1}}}{1-z-z^{2}} = \frac{z}{1-z-z^{2}} + \frac{z}{1-z-z^{2}} \cdot \frac{z^{m+1}}{{(1-z)}^{m+1}} \\
= F(z) + z^{m+1} \frac{1}{{(1-z)}^{m+1}} F(z)
\end{multline*}

\( F(z) = \frac{z}{1-z-z^{2}} \)

\begin{multline*}
a_{n} = [z^{n}]A(z) = F_{n} + [z^{n-m-1}] \frac{1}{{(1-z)}^{m+1}}F(z) \\
= F_{n} + \sum_{k=0}^{n-m-1} \binom{k+m}{m}F_{n-m-1-k}
\end{multline*}

\begin{multline*}
