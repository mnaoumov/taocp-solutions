\section*{2}

We need to prove that \( \forall x \geq 0 \exists m \in \mathbb{Z} \forall k \geq m \exists! \alpha_{k} \in \{ 0, 1 \}: x = \sum_{k}\alpha_{k} \phi^{-k} \)

\[\exists m \in \mathbb{Z} \forall k < m: \alpha_{k} = 0 \tag{1}\label{eq:1}\]

\[\nexists k: \alpha_{k}=\alpha_{k+1} = 1 \tag{2}\label{eq:2}\]

\[\nexists n \forall l\geq 0: \alpha_{n+2l} = 1, \alpha_{n+2l + 1} = 0 \tag{3}\label{eq:3}\]

\section*{Existence}

Let \( x \geq 0 \)

If \( x = 0 \), we take \( \alpha_{k} = 0, k \in \mathbb{Z} \)

If \( x > 0 \exists m \in \mathbb{Z}: \phi^{-m} \leq x < \phi^{-m + 1} \). Let \( \alpha_{m} := 1, \alpha_{k} := 0, k < m \) \( \implies \)~\eqref{eq:1}

\( x_{n} := \sum_{k \leq n} \alpha_{k} \phi^{-k} \)

\( \forall k<m: x_{k}=0, x_{m} = \phi^{-m} \)

\( \forall k \geq m + 1: \alpha_{k} := [x-x_{k-1} \geq \phi^{-k}] \)

Let's prove \( \forall k \geq m: 0 \leq x - x_{k} < \phi^{-k} \quad P(k) \)

\( P(m): \phi^{-m} - \phi^{-m} \leq x - x^{m} < \phi^{-m+1} - \phi^{-m} = \phi^{-m}(\phi - 1) \)

\( \phi^{2} = \phi + 1 \implies \phi = 1 + \phi^{-1} \implies \phi - 1 = \phi^{-1} \)

\( 0 \leq x - x_{m} < \phi^{-m}\cdot\phi^{-1} < \phi^{-m} \)

\( (k \implies k + 1) \)

Assume \( P(k) \implies 0 \leq x - x_{k} < \phi^{k} \)

\( P(k+1): x - x_{k+1} = x-x_{k} - \alpha_{k+1}\phi^{k+1} \)

If \( \alpha_{k+1} = 0 \implies x - x_{k} < \phi^{-(k+1)} \)

\( x_{k+1} = x_{k} \implies 0 \leq x - x_{k+1} = x - x_{k} < \phi^{-(k+1)} \)

If \( \alpha_{k+1} = 1 \implies x - x_{k} \geq \phi^{-(k+1)} \)

\( x - x_{k+1} = x - x_{k} - \phi^{-(k+1)} \geq 0 \)

\( x - x_{k+1} = x - x_{k} - \phi^{-(k+1)} < \phi^{-k} - \phi^{-(k+1)} = \phi^{-(k+1)} (\phi - 1) = \)

\( = \phi^{-(k+1)} \cdot \phi^{-1} < \phi^{-(k+1)} \)

\( \implies P(k+1) \)

\( \implies \lim_{k \to \infty} x_{k} = x \)

\( x = \sum_{k} \alpha_{k} \phi^{-k} \)

Now let's prove~\eqref{eq:2}

Assume \( k = \min \{ s: \alpha_{s} = \alpha_{s+1} = 1 \} \)

As \( k \) is minimal \( \implies \alpha_{k-1} = 0 \)

\[x - x_{k-2} < \phi^{-(k-1)} \tag{4}\label{eq:4}\]

\( x_{k-1} = x_{k-2} \)

\( \alpha_{k} = 1 \implies x - x_{k-1} \geq \phi^{-k} \)

\( x_{k} = x_{k-1} + \phi^{-k} = x_{k-2} + \phi^{-k} \)

\( \alpha_{k+1} = 1 \implies x - x_{k} \geq \phi^{-(k+1)} \)

\( x - (x_{k-2} + \phi^{-k}) \geq \phi^{-(k+1)} \)

\( x - x_{k-2} \geq \phi^{-k} + \phi^{-(k+1)} = \phi^{-(k+1)}(\phi + 1) =  \phi^{-(k+1)} \cdot \phi^{2} = \phi^{-(k-1)} \)

Contradicts to~\eqref{eq:4} \( \implies \perp \implies \)~\eqref{eq:2}

Now let's prove~\eqref{eq:3}

Assume \( \exists n \forall l\geq 0: \alpha_{n+2l} = 1, \alpha_{n+2l + 1} = 0 \)

\( \alpha_{n} = 1 \overset{\eqref{eq:2}}{\implies} \alpha_{n-1} = 0 \)

\[x - x_{n-2} < \phi^{-(n-1)} \tag{5}\label{eq:5}\]

\( x_{n-1} = x_{n-2} \)

\( x = x_{n-1} + \sum_{l \geq 0}\phi^{-(n+2l)} = x_{n-2} + \frac{\phi^{-n}}{1-\phi^{-2}} \)

\( \phi^{2} = \phi + 1 \implies 1 = \phi^{-1} + \phi^{-2} \implies 1 - \phi^{-2} = \phi^{-1} \)

\( x = x_{n-2} + \phi^{-(n-1)} \)

\( x - x_{n-2} = \phi^{-(n-1)} \)

Contradicts to~\eqref{eq:5} \( \implies \perp \implies \)~\eqref{eq:3}

\section*{Uniqueness}

Assume there are different expansions

Let \( x = \sum_{k} \alpha_{k} \phi^{-k} = \sum_{k} \beta_{k} \phi^{-k} \)

\( m = \min \{k: \alpha_{k} \neq \beta_{k} \} \)

For clarity, \( \alpha_{m} = 1, \beta_{m} = 0 \)

\( \forall k<m: \alpha_{k} = \beta_{k} \implies x_{m-1} = \sum_{k<m} \alpha_{k} \phi^{-k} = \sum_{k<m} \beta_{k} \phi^{-k} \)

\( \implies y = x - x_{m-1} = \sum_{k \geq m} \alpha_{k} \phi^{-k} = \sum_{k \geq m} \beta_{k} \phi^{-k} \)

\[y \geq \alpha_{m} \phi^{-m} = \phi^{-m} \tag{6}\label{eq:6}\]

Let \( B := \{ k \geq m + 1: \beta_{k} = 1 \} = \{ k_{1}, k_{2}, \ldots, k_{r} \}, 0 \leq r \leq \infty, k_{1} < k_{2} < \cdots < k_{r} \)

\( y = \sum_{j = 1}^{r} \phi^{-k_{j}} \)

\( \beta_{k_{j}} = 1 \implies \beta_{k_{j}+1} = 0 \implies k_{j+1} \geq k_{j} + 2 \implies k_{j} \geq m + 2j - 1 \)

\( y = \sum_{j = 1}^{r} \phi^{-k_{j}} \leq \sum_{j = 1}^{r} \phi^{-(m-1+2j)} \)

If \( r < \infty \implies y < \sum_{j = 1}^\infty \phi^{-(m-1+2j)} = \frac{\phi^{-(m+1)}}{1-\phi^{-2}} = \phi^{-m} \)

If \( r = \infty \),~\eqref{eq:3} \( \implies y < \sum_{j = 1}^\infty \phi^{-(m-1+2j)} = \phi^{-m} \)

In both cases \( y < \phi^{-m} \), contradicts to~\eqref{eq:6} \( \implies \perp \)

\( \implies \) uniqueness.

\section*{Representation of integers}

\( \phi^{p} = F_{p} \phi + F_{p-1} \)

\( \phi^{-p} = F_{-p} \phi + F_{-p-1} = {(-1)}^{p-1}F_{p} \phi + {(-1)}^{p}F_{p+1} = {(-1)}^{p}(-F_{p}\phi + F_{p+1}) \)

\( \phi^{p} + {(-1)}^{p}\phi^{-p} = F_{p-1} + F_{p+1} =: L_{p} \)

\( L_{p} + L_{p+1} = L_{p+2} \)

\( L_{2p} = \phi^{2p} + \phi^{-2p} \)

\( = \phi^{2p+1} + \phi^{-(2p+2)} - \phi^{-2p} \)
