# 1.2.5-17

${} \prod_{n \geq 1}\prod_{j=1}^k \frac{n + \alpha_j}{n + \beta_j} = \lim_{N\to\infty}\prod_{n=1}^N\prod_{j=1}^k \frac{n + \alpha_j}{n + \beta_j} = \lim_{N\to\infty}\prod_{j=1}^k\frac{(\alpha_j + 1)(\alpha_j + 2) \cdot\ \cdots\ \cdot (\alpha_j + N)}{(\beta_j + 1)(\beta_j + 2) \cdot\ \cdots\ \cdot (\beta_j + N)} = {}$

${} = \lim_{N\to\infty}\prod_{j=1}^k(\frac{(N-1)^{\beta_j+1}(N-1)!}{(\beta_j + 1)((\beta_j + 1) + 1) \cdot\ \cdots\ \cdot ((\beta_j + 1) + (N-1))} \cdot \frac{(\alpha_j + 1)((\alpha_j + 1) + 1) \cdot\ \cdots\ \cdot ((\alpha_j + 1) + (N-1))}{(N-1)^{\alpha_j+1}(N-1)!} \cdot (N-1)^{\alpha_j-\beta_j}) = {}$

${} = \lim_{N\to\infty} \cdot (N-1)^{\sum_{j=1}^k{(\alpha_j-\beta_j)}} \prod_{j=1}^k(\frac{(N-1)^{\beta_j+1}(N-1)!}{(\beta_j + 1)((\beta_j + 1) + 1) \cdot\ \cdots\ \cdot ((\beta_j + 1) + (N-1))} \cdot \frac{(\alpha_j + 1)((\alpha_j + 1) + 1) \cdot\ \cdots\ \cdot ((\alpha_j + 1) + (N-1))}{(N-1)^{\alpha_j+1}(N-1)!}) = {}$

${} = \lim_{N\to\infty} \cdot (N-1)^{\sum_{j=1}^k{\alpha_j}-\sum_{j=1}^k{\beta_j}} \prod_{j=1}^k(\frac{(N-1)^{\beta_j+1}(N-1)!}{(\beta_j + 1)((\beta_j + 1) + 1) \cdot\ \cdots\ \cdot ((\beta_j + 1) + (N-1))} \cdot \frac{(\alpha_j + 1)((\alpha_j + 1) + 1) \cdot\ \cdots\ \cdot ((\alpha_j + 1) + (N-1))}{(N-1)^{\alpha_j+1}(N-1)!}) = {}$

${} \overset{\sum_{j=1}^k{\alpha_j}-\sum_{j=1}^k{\beta_j} = 0}{=} \lim_{N\to\infty} \prod_{j=1}^k(\frac{(N-1)^{\beta_j+1}(N-1)!}{(\beta_j + 1)((\beta_j + 1) + 1) \cdot\ \cdots\ \cdot ((\beta_j + 1) + (N-1))} \cdot \frac{(\alpha_j + 1)((\alpha_j + 1) + 1) \cdot\ \cdots\ \cdot ((\alpha_j + 1) + (N-1))}{(N-1)^{\alpha_j+1}(N-1)!}) = {}$

${} = \prod_{j=1}^k \lim_{N\to\infty} (\frac{(N-1)^{\beta_j+1}(N-1)!}{(\beta_j + 1)((\beta_j + 1) + 1) \cdot\ \cdots\ \cdot ((\beta_j + 1) + (N-1))} \cdot \frac{(\alpha_j + 1)((\alpha_j + 1) + 1) \cdot\ \cdots\ \cdot ((\alpha_j + 1) + (N-1))}{(N-1)^{\alpha_j+1}(N-1)!}) = {}$

${} = \prod_{j=1}^k \frac{\Gamma(\beta_j + 1)}{\Gamma(\alpha_j + 1)} {}$
