\subsubsection*{a)}

$\sqrt[n]{1+x} - 1 \leq \frac{x}{n} \iff 1+x \leq (1+\frac{x}{n})^n$

$y := \frac{x}{n}$

$(1+y)^n \geq 1 + ny$

Prove by induction

$n = 0: (1+y)^0=1=1+0\cdot y$

$(n \implies n + 1)$

$(1+y)^{n+1} = (1+y)^n \cdot (1+y) \geq (1+ny)\cdot(1+y)=1+(n+1)y+y^2 \geq 1+(n+1)y$

\subsubsection*{b)}

$b^{n+\frac{d_1}{10}+\cdots+\frac{d_k}{10^k}} \leq b^x < b^{n+\frac{d_1}{10}+\cdots+\frac{d_k}{10^k} + \frac{1}{10^k}}$

$b^{n+\frac{d_1}{10}+\cdots+\frac{d_k}{10^k} + \frac{1}{10^k}} - b^{n+\frac{d_1}{10}+\cdots+\frac{d_k}{10^k}} = b^{n+\frac{d_1}{10}+\cdots+\frac{d_k}{10^k}} \cdot (b^{\frac{1}{10^k}} - 1) < b^{n+1} \cdot (b^{\frac{1}{10^k}} - 1)$

$x := b - 1, n := 10^k$

$(b^{\frac{1}{10^k}} - 1) = \sqrt[n]{1+x} - 1 \leq \frac{x}{n} = \frac{b-1}{10^k}$
