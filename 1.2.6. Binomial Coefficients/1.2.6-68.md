# 1.2.6-68

${} a_k := \binom{n}{k}p^k(1-p)^{n-k}(k-np) {}$

${} S = \sum_k |a_k| = \sum_{k: k \geq np} a_k - \sum_{k: k < np} a_k = \sum_{k: k \geq \lceil np \rceil} a_k - \sum_{k: k < \lceil np \rceil} a_k = \sum_{k = \lceil np \rceil}^n a_k - \sum_{k = 0}^{\lceil np \rceil - 1} a_k {}$

${} b_k := k \binom{n}{k}p^k(1-p)^{n+1-k} {}$

${} b_k - b_{k+1} = k \binom{n}{k}p^k(1-p)^{n+1-k} - (k+1) \binom{n}{k+1}p^{k+1}(1-p)^{n+1-k-1} = {}$

${} = p^k(1-p)^{n-k} \left(k\binom{n}{k}(1-p) - (k+1)\binom{n}{k+1}p\right) =: A {}$

${} (k+1)\binom{n}{k+1} = (n-k)\binom{n}{k} {}$

${} A = \binom{n}{k}p^k(1-p)^{n-k}(k(1-p) - (n-k)p) = \binom{n}{k}p^k(1-p)^{n-k}(k - np) = a_k {}$

${} m := \lceil np \rceil {}$

${} S = b_m - b_{n+1} - (b_0 - b_m) = 2b_m - b_{n+1} - b_0 {}$

${} b_0 = b_{n+1} = 0 {}$

${} S = 2b_m {}$
