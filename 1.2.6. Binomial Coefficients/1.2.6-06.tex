$\binom{r-1}{k} = \binom{r}{k} - \binom{r-1}{k-1}$

$\binom{r}{k} = 0, k < 0$

$\binom{r}{0} = 1$

$\binom{-1}{1} = \binom{0}{1} - \binom{-1}{0} = 0 - 1 = -1$

$\binom{-1}{2} = \binom{0}{2} - \binom{-1}{1} = 0 - (-1) = 1$

$\binom{-1}{3} = \binom{0}{3} - \binom{-1}{2} = 0 - 1 = -1$

$\binom{-1}{k} = (-1)^k, k \geq 0$

$\binom{-2}{1} = \binom{-1}{1} - \binom{-2}{0} = -1 - 1 = -2$

$\binom{-2}{2} = \binom{-1}{2} - \binom{-2}{1} = 1 - (-2) = 3$

$\binom{-2}{3} = \binom{-1}{3} - \binom{-2}{2} = -1 - 3 = -4$

$\binom{-2}{k} = (-1)^k (k + 1), k \geq 0$

$\binom{-3}{1} = \binom{-2}{1} - \binom{-3}{0} = -2 - 1 = -3$

$\binom{-3}{2} = \binom{-2}{2} - \binom{-3}{1} = 3 - (-3) = 6$

$\binom{-3}{3} = \binom{-2}{3} - \binom{-3}{2} = -4 - 6 = -10$

$\binom{-3}{k} = (-1)^k \frac{(k + 1)(k + 2)}{2}, k \geq 0$
